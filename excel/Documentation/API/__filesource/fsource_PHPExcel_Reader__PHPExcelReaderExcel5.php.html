  290.         /**
  291.          * PARSE WORKBOOK
  292.          *
  293.          **/
  294.         $pos = 0;
  295.         $code = ord($this->_data[$pos]) | ord($this->_data[$pos + 1]) << 8;
  296.         $length = ord($this->_data[$pos + 2]) | ord($this->_data[$pos + 3]) << 8;
  297.         $version = ord($this->_data[$pos + 4]) | ord($this->_data[$pos + 5]) << 8;

  305.             return false;
  306.         }
  307.         $pos += $length + 4;
  308.         $code = ord($this->_data[$pos]) | ord($this->_data[$pos + 1]) << 8;
  309.         $length = ord($this->_data[$pos + 2]) | ord($this->_data[$pos + 3]) << 8;
  310.         $recordData = substr($this->_data, $pos + 4, $length);
  311.         
  312.         while ($code != self::XLS_Type_EOF){
  313.             switch ($code) {
  314.                 case self::XLS_Type_SST:
  315.                     /**
  316.                      * SST - Shared String Table
  317.                      *
  318.                      * This record contains a list of all strings used anywhere
  319.                      * in the workbook. Each string occurs only once. The
  320.                      * workbook uses indexes into the list to reference the
  321.                      * strings.
  322.                      *
  323.                      * --    "OpenOffice.org's Documentation of the Microsoft
  324.                      *         Excel File Format"
  325.                      **/
  326.                     // offset: 0; size: 4; total number of strings

  328.                     $spos = $pos + 4;
  329.                     $limitpos = $spos + $length;
  330.                     $uniqueStrings = $this->_GetInt4d($this->_data, $spos + 4);
  331.                     $spos += 8;

  333.                     for ($i = 0; $i < $uniqueStrings; $i++) {
  334.                         if ($spos == $limitpos) {
  335.                             // then we have reached end of SST record data
  336.                             $opcode = ord($this->_data[$spos]) | ord($this->_data[$spos + 1])<<8;
  337.                             $conlength = ord($this->_data[$spos + 2]) | ord($this->_data[$spos + 3])<<8;

  339.                                 // broken file, something is wrong
  340.                                 return -1;
  341.                             }
  342.                             $spos += 4;
  343.                             $limitpos = $spos + $conlength;
  344.                         }
  345.                         // Read in the number of characters in the Unicode string
  346.                         $numChars = ord($this->_data[$spos]) | (ord($this->_data[$spos + 1]) << 8);
  347.                         $spos += 2;
  348.                         // option flags
  349.                         $optionFlags = ord($this->_data[$spos]);
  350.                         $spos++;
  351.                         // bit: 0; mask: 0x01; 0 = compressed; 1 = uncompressed
  352.                         $asciiEncoding = (($optionFlags & 0x01) == 0) ;
  353.                         // bit: 2; mask: 0x02; 0 = ordinary; 1 = Asian phonetic
  354.                         $extendedString = ( ($optionFlags & 0x04) != 0); // Asian phonetic
  355.                         // bit: 3; mask: 0x03; 0 = ordinary; 1 = Rich-Text
  356.                         $richString = ( ($optionFlags & 0x08) != 0);
  357.                         if ($richString) { // Read in the crun
  358.                             // number of Rich-Text formatting runs
  359.                             $formattingRuns = ord($this->_data[$spos]) | (ord($this->_data[$spos + 1]) << 8);
  360.                             $spos += 2;
  361.                         }
  362.                         if ($extendedString) {
  363.                             // size of Asian phonetic setting
  364.                             $extendedRunLength = $this->_GetInt4d($this->_data, $spos);
  365.                             $spos += 4;
  366.                         }
  367.                         // read in the characters
  368.                         $len = ($asciiEncoding) ? $numChars : $numChars * 2;
  369.                         if ($spos + $len < $limitpos) {
  370.                             $retstr = substr($this->_data, $spos, $len);
  371.                             $spos += $len;
  372.                         } else {
  373.                             // found countinue record
  374.                             $retstr = substr($this->_data, $spos, $limitpos - $spos);
  375.                             $bytesRead = $limitpos - $spos;
  376.                             // remaining characters in Unicode string
  377.                             $charsLeft = $numChars - (($asciiEncoding) ? $bytesRead : ($bytesRead / 2));
  378.                             $spos = $limitpos;
  379.                             // keep reading the characters
  380.                             while ($charsLeft > 0) {
  381.                                 // record data 
  382.                                 $opcode = ord($this->_data[$spos]) | ord($this->_data[$spos + 1]) << 8;
  383.
