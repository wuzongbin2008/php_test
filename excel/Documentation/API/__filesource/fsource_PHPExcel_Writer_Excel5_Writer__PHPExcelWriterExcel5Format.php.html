  263.         $this->_bold           = 0x0190;
  264.         $this->_italic         = 0;
  265.         $this->_color          = 0x7FFF;
  266.         $this->_underline      = 0;
  267.         $this->_font_strikeout = 0;
  268.         $this->_font_outline   = 0;
  269.         $this->_font_shadow    = 0;
  270.         $this->_font_script    = 0;
  271.         $this->_font_family    = 0;
  272.         $this->_font_charset   = 0;
  273.  
  274.         $this->_num_format     = 0;
  275.  
  276.         $this->_hidden         = 0;
  277.         $this->_locked         = 0;
  278.  
  279.         $this->_text_h_align   = 0;
  280.         $this->_text_wrap      = 0;
  281.         $this->_shrink_to_fit  = 0;
  282.         $this->_text_v_align   = 2;
  283.         $this->_text_justlast  = 0;
  284.         $this->_rotation       = 0;
  285.  
  286.         $this->_fg_color       = 0x40;
  287.         $this->_bg_color       = 0x41;

  295.         $this->_diag           = 0;
  296.  
  297.         $this->_bottom_color   = 0x40;
  298.         $this->_top_color      = 0x40;
  299.         $this->_left_color     = 0x40;
  300.         $this->_right_color    = 0x40;

  314.     /**
  315.     * Generate an Excel BIFF XF record (style or cell).
  316.     *
  317.     * @param string $style The type of the XF record ('style' or 'cell').
  318.     * @return string The XF record
  319.     */
  320.     function getXf($style)
  321.     {
  322.         // Set the type of the XF record and some of the attributes.
  323.         if ($style == 'style') {
  324.             $style = 0xFFF5;
  325.         } else {
  326.             $style   = $this->_locked;
  327.             $style  |= $this->_hidden << 1;
  328.         }
  329.  
  330.         // Flags to indicate if attributes have been set.
  331.         $atr_num     = ($this->_num_format != 0)?1:0;
  332.         $atr_fnt     = ($this->font_index != 0)?1:0;
  333.         $atr_alc     = ($this->_text_wrap)?1:0;
  334.         $atr_bdr     = ($this->_bottom   ||
  335.                         $this->_top      ||
  336.                         $this->_left     ||
  337.                         $this->_right)?1:0;
  338.         $atr_pat     = (($this->_fg_color != 0x40) ||
  339.                         ($this->_bg_color != 0x41) ||
  340.                         $this->_pattern)?1:0;
  341.         $atr_prot    = $this->_locked | $this->_hidden;
  342.  
  343.         // Zero the default border colour if the border has not been set.
  344.         if ($this->_bottom == 0) {
  345.             $this->_bottom_color = 0;
  346.         }
  347.         if ($this->_top  == 0) {
  348.             $this->_top_color = 0;
  349.         }
  350.         if ($this->_right == 0) {
  351.             $this->_right_color = 0;
  352.         }
  353.         if ($this->_left == 0) {
  354.             $this->_left_color = 0;
  355.         }
  356.         if ($this->_diag == 0) {
  357.             $this->_diag_color = 0;
  358.         }
  359.  
  360.         $record         = 0x00E0;              // Record identifier
  361.         if ($this->_BIFF_version == 0x0500) {
  362.             $length         = 0x0010;              // Number of bytes to follow
  363.         }
  364.         if ($this->_BIFF_version == 0x0600) {
  365.             $length         = 0x0014;
  366.         }
  367.  
  368.         $ifnt           = $this->font_index;   // Index to FONT record
  369.         $ifmt           = $this->_num_format;  // Index to FORMAT record
  370.         if ($this->_BIFF_version == 0x0500) {
  371.             $align          = $this->_text_h_align;       // Alignment
  372.             $align         |= $this->_text_wrap     << 3;
  373.             $align         |= $this->_text_v_align  << 4;
  374.             $align         |= $this->_text_justlast << 7;
  375.             $align         |= $this->_rotation      << 8;
  376.             $align         |= $atr_num                << 10;
  377.             $align         |= $atr_fnt                << 11;
  378.             $align         |= $atr_alc                << 12;
  379.             $align         |= $atr_bdr                << 13;
  380.             $align         |= $atr_pat                << 14;
  381.             $align         |= $atr_prot               << 15;
  382.  
  383.             $icv            = $this->_fg_color;       // fg and bg pattern colors
  384.             $icv           |= $this->_bg_color      << 7;
  385.  
  386.             $fill           = $this->_pattern;        // Fill and border line style
  387.             $fill          |= $this->_bottom        << 6;
  388.             $fill          |= $this->_bottom_color  << 9;
  389.  
  390.             $border1        = $this->_top;            // Border line style and color
  391.             $border1       |= $this->_left          << 3;
  392.             $border1       |= $this->_right         << 6;
  393.             $border1       |= $this->_top_color     << 9;
  394.  
  395.             $border2        = $this->_left_color;     // Border color
  396.             $border2       |= $this->_right_color   << 7;
  397.  
  398.             $header      = pack("vv",       $record, $length);
  399.             $data        = pack("vvvvvvvv", $ifnt, $ifmt, $style, $align,
  400.                                             $icv, $fill,
  401.                                             $border1, $border2);
  402.         } elseif ($this->_BIFF_version == 0x0600) {
  403.             $align          = $this->_text_h_align;       // Alignment
  404.             $align         |= $this->_text_wrap     << 3;
  405.             $align         |= $this->_text_v_align  << 4;
  406.             $align         |= $this->_text_justlast << 7;
  407.  
  408.             $used_attrib    = $atr_num              << 2;
  409.             $used_attrib   |= $atr_fnt              << 3;
  410.             $used_attrib   |= $atr_alc              << 4;
  411.             $used_attrib   |= $atr_bdr              << 5;
  412.             $used_attrib   |= $atr_pat              << 6;
  413.             $used_attrib   |= $atr_prot             << 7;
  414.  
  415.             $icv            = $this->_fg_color;      // fg and bg pattern colors
  416.             $icv           |= $this->_bg_color      << 7;
  417.  
  418.             $border1        = $this->_left;          // Border line style and color
  419.             $border1       |= $this->_right         << 4;
  420.             $border1       |= $this->_top           << 8;
  421.             $border1       |= $this->_bottom        << 12;
  422.             $border1       |= $this->_left_color    << 16;
  423.             $border1       |= $this->_right_color   << 23;
  424.             $diag_tl_to_rb = 0; // FIXME: add method
  425.             $diag_tr_to_lb = 0; // FIXME: add method
  426.             $border1       |= $diag_tl_to_rb        << 30;
  427.             $border1       |= $diag_tr_to_lb        << 31;
  428.  
  429.             $border2        = $this->_top_color;    // Border color
  430.             $border2       |= $this->_bottom_color   << 7;
  431.             $border2       |= $this->_diag_color     << 14;
  432.             $border2       |= $this->_diag           << 21;
  433.             $border2       |= $this->_pattern        << 26;
  434.  
  435.             $header      = pack("vv",       $record, $length);
  436.  
  437.             $rotation      = 0x00;
  438.  
  439.             //BIFF8 options: identation, shrinkToFit and  text direction
  440.             //currently, only shrinkToFit supported
  441.             //$biff8_options = 0x00;
  442.             $biff8_options = $this->_shrink_to_fit << 4;
  443.             
  444.             $data  = pack("vvvC", $ifnt, $ifmt, $style, $align);
  445.             $data .= pack("CCC", $rotation, $biff8_options, $used_attrib);
  446.             $data .= pack("VVv", $border1, $border2, $icv);
  447.         }
  448.  
  449.         return($header . $data);
  450.     }

  452.     /**
  453.     * Generate an Excel BIFF FONT record.
  454.     *
  455.     * @return string The FONT record
  456.     */
  457.     function getFont()
  458.     {
  459.         $dyHeight   = $this->_size * 20;    // Height of font (1/20 of a point)
  460.         $icv        = $this->_color;        // Index to color palette
  461.         $bls        = $this->_bold;         // Bold style
  462.         $sss        = $this->_font_script;  // Superscript/subscript
  463.         $uls        = $this->_underline;    // Underline
  464.         $bFamily    = $this->_font_family;  // Font family
  465.         $bCharSet   = $this->_font_charset; // Character set
  466.         $encoding   = 0;                    // TODO: Unicode support
  467.  
  468.         $cch        = strlen($this->_font_name); // Length of font name
  469.         $record     = 0x31;                      // Record identifier
  470.         if ($this->_BIFF_version == 0x0500) {
  471.             $length     = 0x0F + $cch;            // Record length
  472.         } elseif ($this->_BIFF_version == 0x0600) {
  473.             $length     = 0x10 + $cch;
  474.         }
  475.         $reserved   = 0x00;                // Reserved
  476.         $grbit      = 0x00;                // Font attributes
  477.         if ($this->_italic) {
  478.             $grbit     |= 0x02;
  479.         }
  480.         if ($this->_font_strikeout) {
  481.             $grbit     |= 0x08;
  482.         }
  483.         if ($this->_font_outline) {
  484.             $grbit     |= 0x10;
  485.         }
  486.         if ($this->_font_shadow) {
  487.             $grbit     |= 0x20;
  488.         }
  489.  
  490.         $header  = pack("vv",         $record, $length);
  491.         if ($this->_BIFF_version == 0x0500) {
  492.             $data    = pack("vvvvvCCCCC", $dyHeight, $grbit, $icv, $bls,
  493.                                           $sss, $uls, $bFamily,
  494.                                           $bCharSet, $reserved, $cch);
  495.         } elseif ($this->_BIFF_version == 0x0600) {
  496.             $data    = pack("vvvvvCCCCCC", $dyHeight, $grbit, $icv, $bls,
  497.                                            $sss, $uls, $bFamily,
  498.                                            $bCharSet, $reserved, $cch, $encoding);
  499.         }
  500.         return($header . $data . $this->_font_name);
  501.     }

  503.     /**
  504.     * Returns a unique hash key for a font.
  505.     * Used by PHPExcel_Writer_Excel5_Workbook::_storeAllFonts()
  506.     *
  507.     * The elements that form the key are arranged to increase the probability of
  508.     * generating a unique key. Elements that hold a large range of numbers
  509.     * (eg. _color) are placed between two binary elements such as _italic

  511.     * @return string A key for this font
  512.     */
  513.     function getFontKey()
  514.     {
  515.         $key  = "$this->_font_name$this->_size";
  516.         $key .= "$this->_font_script$this->_underline";

  534.     /**
  535.     * Used in conjunction with the set_xxx_color methods to convert a color
  536.     * string into a number. Color range is 0..63 but we will restrict it
  537.     * to 8..63 to comply with Gnumeric. Colors 0..7 are repeated in 8..15.
  538.     *
  539.     * @access private
  540.     * @param string $name_color name of the color (i.e.: 'blue', 'red', etc..). Optional.
  541.     * @return integer The color index
  542.     */
  543.     function _getColor($name_color = '')
  544.     {
  545.         $colors = array(
  546.                          'aqua'    => 0x07,
  547.                          'cyan'    => 0x07,
  548.                          'black'   => 0x00,
  549.                          'blue'    => 0x04,
  550.                          'brown'   => 0x10,
  551.                          'magenta' => 0x06,
  552.                          'fuchsia' => 0x06,
  553.                          'gray'    => 0x17,
  554.                          'grey'    => 0x17,
  555.                          'green'   => 0x11,
  556.                          'lime'    => 0x03,
  557.                          'navy'    => 0x12,
  558.                          'orange'  => 0x35,
  559.                          'purple'  => 0x14,
  560.                          'red'     => 0x02,
  561.                          'silver'  => 0x16,
  562.                          'white'   => 0x01,
  563.                          'yellow'  => 0x05
  564.                       );
  565.  
  566.         // Return the default color, 0x7FFF, if undef,
  567.         if ($name_color == '') {
  568.             return(0x7FFF);
  569.         }
  570.  
  571.         // or the color string converted to an integer,
  572.         if (isset($colors[$name_color])) {
  573.             return($colors[$name_color]);
  574.         }
  575.  
  576.         // or the default color if string is unrecognised,
  577.         if (preg_match("/\D/",$name_color)) {
  578.             return(0x7FFF);
  579.         }
  580.  
  581.         // or an index < 8 mapped into the correct range,
  582.         //if ($name_color < 8) {
  583.         //    return($name_color + 8);
  584.         //}
  585.  
  586.         // or the default color if arg is outside range,
  587.         if ($name_color > 63) {
  588.             return(0x7FFF);
  589.         }
  590.  
  591.         // or an integer in the valid range
  592.         return($name_color);
  593.     }
