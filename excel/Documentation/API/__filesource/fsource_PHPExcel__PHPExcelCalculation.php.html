  518.     /**
  519.      * Calculate cell value (using formula)
  520.      *
  521.      * @param    PHPExcel_Cell    $pCell    Cell to calculate
  522.      * @return    mixed 
  523.      * @throws    Exception
  524.      */
  525.     public function calculate(PHPExcel_Cell $pCell = null) {
  526.         // Return value
  527.         $returnValue = '';
  528.  
  529.         // Is the value present in calculation cache?
  530.         if ($this->getCalculationCacheEnabled ()) {
  531.             if (isset ( $this->_calculationCache [$pCell->getParent ()->getTitle ()] [$pCell->getCoordinate ()] )) {
  532.                 if ((time () + microtime ()) - $this->_calculationCache [$pCell->getParent ()->getTitle ()] [$pCell->getCoordinate ()] ['time'] < $this->_calculationCacheExpirationTime) {
  533.                     return $this->_calculationCache [$pCell->getParent ()->getTitle ()] [$pCell->getCoordinate ()] ['data'];
  534.                 } else {
  535.                     unset ( $this->_calculationCache [$pCell->getParent ()->getTitle ()] [$pCell->getCoordinate ()] );
  536.                 }
  537.             }
  538.         }
  539.  
  540.         // Formula
  541.         $formula = $pCell->getValue ();
  542.  
  543.         // Executable formula array
  544.         $executableFormulaArray = array ( );
  545.  
  546.         // Parse formula into a tree of tokens
  547.         $objParser = new PHPExcel_Calculation_FormulaParser ( $formula );
  548.  
  549.         // Loop trough parsed tokens and create an executable formula
  550.         $inFunction = false;
  551.         $token = null;
  552.         for($i = 0; $i < $objParser->getTokenCount (); $i ++) {
  553.             $token = $objParser->getToken ( $i );
  554.  
  555.             // Is it a cell reference? Not in a function?
  556.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_RANGE) && (strpos ( $token->getValue (), ':' ) === false) && (! $inFunction)) {
  557.                 // Adjust reference
  558.                 $reference = str_replace ( '$', '', $token->getValue () );
  559.  
  560.                 // Get value
  561.                 $calculatedValue = null;
  562.                 if ($pCell->getParent ()->getCell ( $reference )->getValue () instanceof PHPExcel_RichText) {
  563.                     $calculatedValue = $pCell->getParent ()->getCell ( $reference )->getValue ()->getPlainText ();
  564.                 } else {
  565.                     $calculatedValue = $pCell->getParent ()->getCell ( $reference )->getCalculatedValue ();
  566.                 }
  567.                 if (is_string ( $calculatedValue )) {
  568.                     $calculatedValue = '"' . $calculatedValue . '"';
  569.                 }
  570.  
  571.                 // Add to executable formula array
  572.                 array_push ( $executableFormulaArray, $calculatedValue );
  573.  
  574.                 continue;
  575.             }
  576.  
  577.             // Is it a cell reference? In a function?
  578.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_RANGE) && ($inFunction)) {
  579.                 // Adjust reference
  580.                 $reference = str_replace ( '$', '', $token->getValue () );
  581.  
  582.                 // Add to executable formula array
  583.                 array_push ( $executableFormulaArray, '$this->extractRange("' . $reference . '", $pCell->getParent())' );
  584.  
  585.                 continue;
  586.             }
  587.  
  588.             // Is it a concatenation operator?
  589.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERATORINFIX) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_CONCATENATION)) {
  590.                 // Add to executable formula array
  591.                 array_push ( $executableFormulaArray, '.' );
  592.  
  593.                 continue;
  594.             }
  595.  
  596.             // Is it a logical operator?
  597.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERATORINFIX) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_LOGICAL)) {
  598.                 // Temporary variable
  599.                 $tmp = '';
  600.                 switch ( $token->getValue ()) {
  601.                     case '=' :
  602.                         $tmp = '==';
  603.                     break;
  604.                     case '<>' :
  605.                         $tmp = '!=';
  606.                     break;
  607.                     default :
  608.                         $tmp = $token->getValue ();
  609.                 }
  610.  
  611.                 // Add to executable formula array
  612.                 array_push ( $executableFormulaArray, $tmp );
  613.  
  614.                 continue;
  615.             }
  616.  
  617.             // Is it a subexpression?
  618.             if ($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_SUBEXPRESSION) {
  619.                 // Temporary variable
  620.                 $tmp = '';
  621.                 switch ( $token->getTokenSubType ()) {
  622.                     case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START :
  623.                         $tmp = '(';
  624.                     break;
  625.                     case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP :
  626.                         $tmp = ')';
  627.                     break;
  628.                 }
  629.  
  630.                 // Add to executable formula array
  631.                 array_push ( $executableFormulaArray, $tmp );
  632.  
  633.                 continue;
  634.             }
  635.  
  636.             // Is it a function?
  637.             if ($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_FUNCTION) {
  638.                 // Temporary variable
  639.                 $tmp = '';
  640.  
  641.                 // Check the function type
  642.                 if ($token->getValue () == 'ARRAY' || $token->getValue () == 'ARRAYROW') {
  643.                     // An array or an array row...
  644.                     $tmp = 'array(';
  645.                 } else {
  646.                     // A regular function call...
  647.                     switch ( $token->getTokenSubType ()) {
  648.                         case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_START :
  649.                             // Check if the function call is allowed...
  650.                             if (! isset ( $this->_functionMappings [strtoupper ( $token->getValue () )] )) {
  651.                                 return '#NAME?';
  652.                             }
  653.  
  654.                             // Map the function call
  655.                             $tmp = $this->_functionMappings [strtoupper ( $token->getValue () )]->getPHPExcelName () . '(';
  656.                             $inFunction = true;
  657.                         break;
  658.                         case PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_STOP :
  659.                             $tmp = ')';
  660.                         break;
  661.                     }
  662.                 }
  663.  
  664.                 // Add to executable formula array
  665.                 array_push ( $executableFormulaArray, $tmp );
  666.  
  667.                 continue;
  668.             }
  669.  
  670.             // Is it text?
  671.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_TEXT)) {
  672.                 // Add to executable formula array
  673.                 array_push ( $executableFormulaArray, '"' . $token->getValue () . '"' );
  674.  
  675.                 continue;
  676.             }
  677.  
  678.             // Is it a number?
  679.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_NUMBER)) {
  680.                 // Add to executable formula array
  681.                 array_push ( $executableFormulaArray, $token->getValue () );
  682.  
  683.                 continue;
  684.             }
  685.  
  686.             // Is it an error? Add it as text...
  687.             if (($token->getTokenType () == PHPExcel_Calculation_FormulaToken::TOKEN_TYPE_OPERAND) && ($token->getTokenSubType () == PHPExcel_Calculation_FormulaToken::TOKEN_SUBTYPE_ERROR)) {
  688.                 // Add to executable formula array
  689.                 array_push ( $executableFormulaArray, '"' . $token->getValue () . '"' );
  690.  
  691.                 continue;
  692.             }
  693.  
  694.             // Is it something else?
  695.             array_push ( $executableFormulaArray, $token->getValue () );
  696.         }
  697.  
  698.         // Evaluate formula
  699.         try {
  700.             $formula = implode ( ' ', $executableFormulaArray );
  701.             $formula = str_replace ( '(,', '(null,', $formula );
  702.             $formula = str_replace ( ',,', ',null,', $formula );
  703.             $formula = str_replace ( ',)', ',null)', $formula );
  704.  
  705.             $formula = str_replace ( '$this', '$pThat', $formula );
  706.
